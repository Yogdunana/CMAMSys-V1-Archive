```typescript
'use client';

import React, { useState, useEffect } from 'react';
import ReactMarkdown from 'react-markdown';
import remarkGfm from 'remark-gfm';
import rehypeHighlight from 'rehype-highlight';
import 'highlight.js/styles/github.css';

export default function DocPreviewPage({ params }: { params: Promise<{ path: string[] }> }) {
  const [content, setContent] = useState<string>('');
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    async function loadDoc() {
      const resolvedParams = await params;
      const docPath = resolvedParams.path.join('/');

      try {
        const response = await fetch(`/api/docs/content?path=${docPath}`);
        const data = await response.json();

        if (data.success) {
          setContent(data.content);
        } else {
          setError(data.error.message);
        }
      } catch (err) {
        setError('Failed to load document');
      } finally {
        setLoading(false);
      }
    }

    loadDoc();
  }, [params]);

  if (loading) {
    return (
      <div className="min-h-screen flex items-center justify-center">
        <div className="animate-spin rounded-full h-12 w-12 border-b-2 border-blue-600"></div>
      </div>
    );
  }

  if (error) {
    return (
      <div className="min-h-screen flex items-center justify-center">
        <div className="text-red-600">{error}</div>
      </div>
    );
  }

  return (
    <div className="min-h-screen bg-gray-50">
      <div className="max-w-4xl mx-auto px-4 py-8">
        <div className="bg-white rounded-lg shadow-lg p-8">
          <ReactMarkdown
            remarkPlugins={[remarkGfm]}
            rehypePlugins={[rehypeHighlight]}
            className="prose prose-lg max-w-none"
          >
            {content}
          </ReactMarkdown>
        </div>
      </div>
    </div>
  );
}
```

创建 `src/app/api/docs/content/route.ts`：

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { promises as fs } from 'fs';
import path from 'path';

const DOCS_DIR = path.join(process.cwd(), 'docs');

export async function GET(request: NextRequest) {
  const { searchParams } = new URL(request.url);
  const docPath = searchParams.get('path');

  if (!docPath) {
    return NextResponse.json(
      {
        success: false,
        error: {
          code: 'INVALID_REQUEST',
          message: 'Path parameter is required',
        },
      },
      { status: 400 }
    );
  }

  try {
    const filePath = path.join(DOCS_DIR, `${docPath}.md`);

    // 安全检查：确保文件在 docs 目录内
    const normalizedPath = path.normalize(filePath);
    if (!normalizedPath.startsWith(DOCS_DIR)) {
      return NextResponse.json(
        {
          success: false,
          error: {
            code: 'INVALID_PATH',
            message: 'Invalid document path',
          },
        },
        { status: 403 }
      );
    }

    const content = await fs.readFile(normalizedPath, 'utf-8');

    return NextResponse.json({
      success: true,
      data: {
        content,
        path: docPath,
      },
    });
  } catch (error) {
    return NextResponse.json(
      {
        success: false,
        error: {
          code: 'DOCUMENT_NOT_FOUND',
          message: 'Document not found',
        },
      },
      { status: 404 }
    );
  }
}
```

#### 3.2 创建搜索服务

创建 `src/services/doc-search.ts`：

```typescript
import Fuse from 'fuse.js';

interface DocItem {
  title: string;
  content: string;
  path: string;
  tags?: string[];
}

export class DocSearchService {
  private fuse: Fuse<DocItem>;
  private docs: DocItem[] = [];

  constructor() {
    this.fuse = new Fuse([], {
      keys: ['title', 'content', 'tags'],
      threshold: 0.3,
      includeScore: true,
      includeMatches: true,
    });
  }

  async loadDocuments() {
    // 从 API 加载所有文档
    const response = await fetch('/api/docs/list');
    const data = await response.json();

    if (data.success) {
      this.docs = [];
      for (const doc of data.data) {
        const contentResponse = await fetch(`/api/docs/content?path=${doc.path}`);
        const contentData = await contentResponse.json();

        if (contentData.success) {
          this.docs.push({
            title: doc.title,
            content: contentData.data.content,
            path: doc.path,
            tags: [],
          });
        }
      }

      this.fuse.setCollection(this.docs);
    }
  }

  search(query: string): DocItem[] {
    if (!query.trim()) {
      return this.docs;
    }

    const results = this.fuse.search(query);
    return results.map((result) => result.item);
  }
}

export const docSearchService = new DocSearchService();
```

```typescript
'use client';

import React, { useState, useEffect } from 'react';
import { Search } from 'lucide-react';

interface SearchResult {
  title: string;
  content: string;
  path: string;
}

interface SearchBarProps {
  onSearch: (query: string) => void;
}

export function SearchBar({ onSearch }: SearchBarProps) {
  const [query, setQuery] = useState('');
  const [results, setResults] = useState<SearchResult[]>([]);
  const [isOpen, setIsOpen] = useState(false);

  useEffect(() => {
    const delayDebounceFn = setTimeout(async () => {
      if (query.trim()) {
        const response = await fetch(`/api/docs/search?q=${encodeURIComponent(query)}`);
        const data = await response.json();
        if (data.success) {
          setResults(data.data.results);
        }
      } else {
        setResults([]);
      }
    }, 300);

    return () => clearTimeout(delayDebounceFn);
  }, [query]);

  const handleSubmit = (e: React.FormEvent) => {
    e.preventDefault();
    onSearch(query);
    setIsOpen(false);
  };

  return (
    <div className="relative w-full max-w-xl">
      <form onSubmit={handleSubmit}>
        <div className="relative">
          <input
            type="text"
            value={query}
            onChange={(e) => {
              setQuery(e.target.value);
              setIsOpen(true);
            }}
            onFocus={() => setIsOpen(true)}
            placeholder="Search documentation..."
            className="w-full px-4 py-2 pl-10 border border-gray-300 rounded-lg focus:ring-2 focus:ring-blue-500 focus:border-transparent"
          />
          <Search className="absolute left-3 top-2.5 h-5 w-5 text-gray-400" />
```

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { prisma } from '@/lib/db';

export async function POST(request: NextRequest) {
  try {
    const body = await request.json();
    const { docPath, userId } = body;

    const ipAddress = request.headers.get('x-forwarded-for') ||
                      request.headers.get('x-real-ip') ||
                      'unknown';
    const userAgent = request.headers.get('user-agent') || undefined;

    await prisma.docView.create({
      data: {
        docPath,
        userId: userId || undefined,
        ipAddress,
        userAgent,
      },
    });

    return NextResponse.json({
      success: true,
      message: 'Document view recorded',
    });
  } catch (error) {
    return NextResponse.json(
      {
        success: false,
        error: {
          code: 'RECORD_ERROR',
          message: 'Failed to record document view',
        },
      },
      { status: 500 }
    );
  }
}

export async function GET(request: NextRequest) {
  try {
    const { searchParams } = new URL(request.url);
    const docPath = searchParams.get('docPath');
    const days = parseInt(searchParams.get('days') || '7');

    const startDate = new Date();
    startDate.setDate(startDate.getDate() - days);

    const views = await prisma.docView.findMany({
      where: {
        docPath: docPath || undefined,
        viewedAt: {
          gte: startDate,
        },
      },
      orderBy: {
        viewedAt: 'desc',
      },
    });

    const stats = {
      totalViews: views.length,
      uniqueVisitors: new Set(views.map((v) => v.ipAddress)).size,
      dailyViews: views.reduce((acc, view) => {
        const date = view.viewedAt.toISOString().split('T')[0];
        acc[date] = (acc[date] || 0) + 1;
        return acc;
      }, {} as Record<string, number>),
    };

    return NextResponse.json({
      success: true,
      data: {
        views,
        stats,
      },
    });
  } catch (error) {
    return NextResponse.json(
      {
        success: false,
        error: {
          code: 'FETCH_ERROR',
          message: 'Failed to fetch analytics',
        },
      },
      { status: 500 }
    );
  }
}
```

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { prisma } from '@/lib/db';

export async function POST(request: NextRequest) {
  try {
    const body = await request.json();
    const { query, resultsCount } = body;

    const ipAddress = request.headers.get('x-forwarded-for') ||
                      request.headers.get('x-real-ip') ||
                      'unknown';

    await prisma.docSearch.create({
      data: {
        query,
        resultsCount,
        ipAddress,
      },
    });

    return NextResponse.json({
      success: true,
      message: 'Search recorded',
    });
  } catch (error) {
    return NextResponse.json(
      {
        success: false,
        error: {
          code: 'RECORD_ERROR',
          message: 'Failed to record search',
        },
      },
      { status: 500 }
    );
  }
}

export async function GET(request: NextRequest) {
  try {
    const { searchParams } = new URL(request.url);
    const days = parseInt(searchParams.get('days') || '7');

    const startDate = new Date();
    startDate.setDate(startDate.getDate() - days);

    const searches = await prisma.docSearch.findMany({
      where: {
        searchedAt: {
          gte: startDate,
        },
      },
      orderBy: {
        searchedAt: 'desc',
      },
      take: 100,
    });

    const topQueries = searches.reduce((acc, search) => {
      acc[search.query] = (acc[search.query] || 0) + 1;
      return acc;
    }, {} as Record<string, number>);

    const sortedTopQueries = Object.entries(topQueries)
      .sort((a, b) => b[1] - a[1])
      .slice(0, 10)
      .map(([query, count]) => ({ query, count }));

    return NextResponse.json({
      success: true,
      data: {
        searches,
        topQueries: sortedTopQueries,
      },
    });
  } catch (error) {
    return NextResponse.json(
      {
        success: false,
        error: {
          code: 'FETCH_ERROR',
          message: 'Failed to fetch search analytics',
        },
      },
      { status: 500 }
    );
  }
}
```

```typescript
      if (data.success) {
        setStats(data.data.stats);
      }
      setLoading(false);
    }

    fetchStats();
  }, []);

  if (loading) {
    return <div>Loading...</div>;
  }

  const chartData = Object.entries(stats.dailyViews || {}).map(([date, count]) => ({
    date,
    views: count,
  }));

  return (
    <div className="min-h-screen bg-gray-50">
      <div className="max-w-7xl mx-auto px-4 py-8">
        <h1 className="text-3xl font-bold text-gray-900 mb-8">
          Documentation Analytics
        </h1>

        <div className="grid grid-cols-1 md:grid-cols-3 gap-6 mb-8">
          <div className="bg-white rounded-lg shadow p-6">
            <h3 className="text-lg font-semibold text-gray-900 mb-2">
              Total Views
            </h3>
            <p className="text-3xl font-bold text-blue-600">
              {stats.totalViews}
            </p>
          </div>

          <div className="bg-white rounded-lg shadow p-6">
            <h3 className="text-lg font-semibold text-gray-900 mb-2">
              Unique Visitors
            </h3>
            <p className="text-3xl font-bold text-green-600">
              {stats.uniqueVisitors}
            </p>
          </div>

          <div className="bg-white rounded-lg shadow p-6">
            <h3 className="text-lg font-semibold text-gray-900 mb-2">
              Average Daily Views
            </h3>
            <p className="text-3xl font-bold text-purple-600">
              {(stats.totalViews / 7).toFixed(1)}
            </p>
          </div>
        </div>

        <div className="bg-white rounded-lg shadow p-6">
          <h2 className="text-xl font-semibold text-gray-900 mb-4">
            Daily Views (Last 7 Days)
          </h2>
          <ResponsiveContainer width="100%" height={300}>
            <BarChart data={chartData}>
              <CartesianGrid strokeDasharray="3 3" />
              <XAxis dataKey="date" />
              <YAxis />
              <Tooltip />
              <Bar dataKey="views" fill="#3B82F6" />
            </BarChart>
          </ResponsiveContainer>
        </div>
      </div>
    </div>
  );
}
```
